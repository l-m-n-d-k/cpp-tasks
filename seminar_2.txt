//task 2.1
#include <iostream>
#include <cmath>
using namespace std;

int main() {
  double x, a;
  cout << "write x and a:" << endl;
  cin >> x >> a;

  if (abs(x) < 1) {
    if (x == 0) {
      cout << "error 1" << endl;
    } else {
      cout << "w = " << a * log(abs(x)) << endl;
    }
  } else {
    if (a - pow(x, 2) < 0) {
      cout << "error 2" << endl;
    } else {
      cout << "w = " << sqrt(a - pow(x, 2)) << endl;
    }
  }

  return 0;
}

//task 2.2
#include <iostream>
#include <cmath>
using namespace std;

int main() {
  double x, y, b;
  cout << "write x, y, and b" << endl;
  cin >> x >> y >> b;
    
  if (b - x == 0) {
      cout << "error: division by zero (b - x = 0)" << endl;
      return 1;
  }
    
  if (b - y <= 0) {
      cout << "error: logarithm argument must be positive (b - y <= 0)" << endl;
      return 1;
  }
    
  double z = log(b - y) / (b - x);
    
  cout << "z = " << z << endl;
    
  return 0;
}
